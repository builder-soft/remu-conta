DROP PROCEDURE if exists bsframework.getDomainAttribute;
DROP PROCEDURE if exists bsframework.pGetDomainAttribute;
DELIMITER $$

CREATE PROCEDURE bsframework.pGetDomainAttribute(IN domain BIGINT)
BEGIN
	SELECT		da.cId /*, da.cKey, da.cName, da.cValue*/
	FROM 		bsframework.tDomainAttribute AS da
	WHERE		da.cDomain = domain;
END$$

DELIMITER ;
