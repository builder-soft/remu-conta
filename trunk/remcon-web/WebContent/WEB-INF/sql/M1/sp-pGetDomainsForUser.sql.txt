DROP PROCEDURE if exists bsframework.getDomainsForUser;
DROP PROCEDURE if exists bsframework.pGetDomainsForUser;
DELIMITER $$

CREATE PROCEDURE bsframework.pGetDomainsForUser(IN user BIGINT)
BEGIN
	SELECT		d.cId 
	FROM 		bsframework.tDomain AS d
	LEFT JOIN	tR_UserDomain AS r ON r.cDomain = d.cId
	WHERE		r.cUser = user;
END$$

DELIMITER ;
