DROP PROCEDURE if exists pOpenPeriod;
DELIMITER $$

CREATE PROCEDURE pOpenPeriod(IN pPeriod BIGINT)
/*
1.- se abre / crea un periodo
2.- se ingresan parametros para el calculo
3.- se calcula
4.- se pueden modificar parametros para calcular de nuevo
5.- se calcula con correcciones
6.- se cierra periodo
7.- no se puede volver a calcular nuevamente para ese per√≠odo...
*/
BEGIN
	UPDATE		tPeriod
	SET			cPeriodStatus = 1;
	
	UPDATE		tPeriod
	SET			cPeriodStatus = 2
	WHERE		cId = pPeriod;

END$$

DELIMITER ;
