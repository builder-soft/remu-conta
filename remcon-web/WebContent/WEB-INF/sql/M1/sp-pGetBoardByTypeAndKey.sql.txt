DROP PROCEDURE if exists remcon.pGetBoardByTypeAndKey;
DELIMITER $$

CREATE PROCEDURE remcon.pGetBoardByTypeAndKey(IN pType VARCHAR(15), IN pKey VARCHAR(15))
BEGIN
	SELECT		cId, cType, cKey, cValue, cEnable 
	FROM 		remcon.tBoard
	WHERE		cType = pType AND cKey = pKey
	ORDER BY	cId;
END$$

DELIMITER ;
